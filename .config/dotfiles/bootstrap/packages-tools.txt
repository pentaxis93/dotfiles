# ============================================================================
# MODERN CLI REPLACEMENTS
# ============================================================================
# Better versions of standard Unix tools - all actively used

ripgrep         # Fast grep alternative (rg) - used by Claude Code
fd              # Fast find alternative - better file searching
fzf             # Fuzzy finder - checked for in Fish config
bat             # Cat with syntax highlighting - better file viewing
eza             # Modern ls with git integration - better directory listing

# ============================================================================
# TERMINAL TOOLS
# ============================================================================
# Enhance terminal workflows

tmux            # Terminal multiplexer - split panes, sessions
lazygit         # Terminal UI for git - used by lazydots function
zoxide          # Smart directory jumper - z command learns your habits

# ============================================================================
# DEVELOPMENT UTILITIES
# ============================================================================
# Tools for development workflows

nodejs          # JavaScript runtime - required for Claude Code
npm             # Node package manager - installs Claude Code
gh              # GitHub CLI - interact with GitHub from terminal

# ============================================================================
# ARCHIVE UTILITIES
# ============================================================================
# Handle compressed files

unzip           # Extract ZIP archives - common archive format
p7zip           # 7-Zip support - handles .7z and other formats
